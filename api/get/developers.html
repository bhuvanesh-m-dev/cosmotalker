02 JAVASCRIPT — BROWSER
fetch API 
async function getSpaceInfo(question) {
  try {
    const response = await fetch(
      `https://bhuvanesh-m-dev.github.io/cosmotalker/api/get?q=${encodeURIComponent(question)}`
    );
    const data = await response.json();

    if (data.success) {
      return data.answer;
    } else {
      throw new Error(data.error || "No information found");
    }
  } catch (error) {
    console.error("API Error:", error);
  }
}

// Usage
getSpaceInfo("iss location").then(answer => {
  document.getElementById("result").innerText = answer;
});

04 PYTHON
requests library 
import requests

def get_space_info(question):
    try:
        response = requests.get(
            "https://bhuvanesh-m-dev.github.io/cosmotalker/api/get",
            params={"q": question},
            timeout=30
        )
        response.raise_for_status()
        data = response.json()

        if data.get("success"):
            return data.get("answer")
        else:
            return f"Error: {data.get('error')}"

    except requests.exceptions.RequestException as e:
        return f"Connection error: {e}"

# Usage
answer = get_space_info("iss location")
print(answer)

<?php
function askCosmoTalker($question) {
    $params = http_build_query(['q' => $question]);
    $url = "https://bhuvanesh-m-dev.github.io/cosmotalker/api/get?" . $params;

    $ch = curl_init();
    curl_setopt($ch, CURLOPT_URL, $url);

    $response = curl_exec($ch);
    curl_close($ch);

    return json_decode($response, true);
}

07 JAVA
HttpClient (Java 11+) 
import java.net.http.*;
import java.net.*;
import java.net.URLEncoder;
import java.nio.charset.StandardCharsets;

public class CosmoTalkerAPI {
    private static final HttpClient client = HttpClient.newHttpClient();

    public static String ask(String question) throws Exception {
        String encoded = URLEncoder.encode(question, StandardCharsets.UTF_8);
        String url = "https://bhuvanesh-m-dev.github.io/cosmotalker/api/get?q=" + encoded;

        HttpRequest request = HttpRequest.newBuilder()
            .uri(URI.create(url))
            .GET()
            .build();

        HttpResponse<String> response = client.send(
            request, HttpResponse.BodyHandlers.ofString()
        );
        return response.body();
    }
}
08 C# / .NET
HttpClient async 
using System.Net.Http;

class CosmoTalkerAPI {
    static readonly HttpClient client = new HttpClient();

    public static async Task<string> Ask(string question) {
        string url = $"https://bhuvanesh-m-dev.github.io/cosmotalker/api/get?q={Uri.EscapeDataString(question)}";
        HttpResponseMessage response = await client.GetAsync(url);
        return await response.Content.ReadAsStringAsync();
    }

    static async Task Main() {
        string result = await Ask("iss location");
        Console.WriteLine(result);
    }
}

10 GO
net/http 
package main

import (
    "encoding/json"
    "fmt"
    "io"
    "net/http"
    "net/url"
)

type Response struct {
    Success bool   `json:"success"`
    Answer  string `json:"answer"`
}

func askCosmoTalker(question string) (*Response, error) {
    params := url.Values{}
    params.Add("q", question)
    fullURL := fmt.Sprintf("%s?%s",
        "https://bhuvanesh-m-dev.github.io/cosmotalker/api/get",
        params.Encode())

    resp, _ := http.Get(fullURL)
    defer resp.Body.Close()

    body, _ := io.ReadAll(resp.Body)
    var result Response
    json.Unmarshal(body, &result)
    return &result, nil
}
